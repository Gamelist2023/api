サンプルコード

以下のコードはこのサイトを利用するためのサンプルコードです：

Python

import requests def callFlaskAPI(userMessage): url = "https://2d773fb3-8162-49ea-a4f0-149615c666e7-00-dk8vul44wy2t.pike.replit.dev/ask" params = {"text": userMessage} response = requests.get(url, params=params) unicodeResponse = response.json() utf8Response = unicodeToUtf8(unicodeResponse) return utf8Response def unicodeToUtf8(unicode): return unicode.encode('utf-8')
